# set Address
MOVW R4, 0
MOVT R4, 0x3F20

#Register
ADD R2, R4, 0x08 
LDR R3, (R2)
ORR R3, R3, 0x00000008
STR R3, (R2)

#TurnOn
ADD R3, R4, 0x1C $Repeat
MOVW R2, 0x0000
MOVT R2, 0x0020
STR R2, (R3)

#Delay 7 - 2 = 5
BL :DelayMe

#TurnOff
ADD R3, R4, 0x28
MOVW R2, 0x0000
MOVT R2, 0x0020
STR R2, (R3)

#Delay
BL :DelayMe

B :Repeat

#Delay
MOVW R10, 0x4240 $DelayMe
MOVT R10, 0xF
SUBS R10, R10, 1 $LoopStart

#-1 - 2 = -3
BPL :LoopStart

#Loop
BX R14
